TC05 — CROSS-ENTITY FEE-BPS & THRESHOLD DIFFER (SG vs HK vs CH)
----------------------------------------------------------------------
ASSERTION (bash arithmetic, using captured output files from TC01-TC03):

# Extract numeric fee values and assert ordering: SG(50) < HK(60) < CH(80)
SG_FEE=$(grep 'MGMT-FEE-BPS' tc01.out | grep -oP '[0-9]{3}' | head -1)
HK_FEE=$(grep 'MGMT-FEE-BPS' tc02.out | grep -oP '[0-9]{3}' | head -1)
CH_FEE=$(grep 'MGMT-FEE-BPS' tc03.out | grep -oP '[0-9]{3}' | head -1)
[ "$SG_FEE" = '050' ] || echo FAIL-TC05-sg-fee-value
[ "$HK_FEE" = '060' ] || echo FAIL-TC05-hk-fee-value
[ "$CH_FEE" = '080' ] || echo FAIL-TC05-ch-fee-value
[ "$SG_FEE" -lt "$HK_FEE" ] || echo FAIL-TC05-sg-lt-hk
[ "$HK_FEE" -lt "$CH_FEE" ] || echo FAIL-TC05-hk-lt-ch

# Threshold ordering: SG(250000) < HK(1000000) < CH(5000000)
SG_THR=$(grep 'LARGE-POS-THRESHOLD' tc01.out | grep -oP '[0-9]{9}' | head -1)
HK_THR=$(grep 'LARGE-POS-THRESHOLD' tc02.out | grep -oP '[0-9]{9}' | head -1)
CH_THR=$(grep 'LARGE-POS-THRESHOLD' tc03.out | grep -oP '[0-9]{9}' | head -1)
[ "$SG_THR" = '000250000' ] || echo FAIL-TC05-sg-threshold
[ "$HK_THR" = '001000000' ] || echo FAIL-TC05-hk-threshold
[ "$CH_THR" = '005000000' ] || echo FAIL-TC05-ch-threshold
PASS CRITERIA: zero FAIL lines printed.

TC06 — SECRET REFERENCES CONTAIN NO PLAINTEXT — ONLY KEYVAULT REFS
----------------------------------------------------------------------
SETUP: grep source + output for plaintext secret patterns.

# Source-level check (no plaintext passwords/keys in .cbl or .cpy):
grep -rniE '(password|passwd|secret\s*=\s*[^$]|api.?key\s*=\s*[^$])' \
  cobol/PORTVAL.cbl cobol/ENTITY-COPY.cpy && echo FAIL-TC06-plaintext-in-source

# Runtime output must contain KeyVault reference pattern, not raw value:
for OUT in tc01.out tc02.out tc03.out; do
  grep -q '\${KEYVAULT:wealth-' "$OUT" || echo FAIL-TC06-no-keyvault-ref-in-$OUT
  grep -iE 'password=.{4}|api_key=.{4}' "$OUT" && echo FAIL-TC06-plaintext-in-$OUT
done
