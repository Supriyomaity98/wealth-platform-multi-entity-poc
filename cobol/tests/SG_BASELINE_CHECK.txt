*> ================================================================
      *> SG_BASELINE_CHECK.txt - PORTVAL entity-context sanity check
      *> Executable via GnuCOBOL:
      *>   cobc -x -free cobol/tests/SG_BASELINE_CHECK.cob
      *>   ./SG_BASELINE_CHECK
      *> Exit 0 = all pass.  Exit 1 = one or more assertions failed.
      *> ================================================================
       IDENTIFICATION DIVISION.
       PROGRAM-ID. SG-BASELINE-CHECK.

       DATA DIVISION.
       WORKING-STORAGE SECTION.
       01  WS-ENTITY-ID          PIC X(02).
       01  WS-CURRENCY           PIC X(03).
       01  WS-LOCALE             PIC X(05).
       01  WS-REGULATOR          PIC X(08).
       01  WS-BOOKING-CENTRE     PIC X(16).
       01  WS-MGMT-FEE-BPS       PIC 9(03).
       01  WS-LARGE-POS-THRESH   PIC 9(09)V99.
       01  WS-SUITABILITY        PIC X(20).
       01  WS-VAULT-REF          PIC X(60).
       01  WS-PASS-COUNT         PIC 9(02) VALUE ZERO.
       01  WS-FAIL-COUNT         PIC 9(02) VALUE ZERO.
       01  WS-TEST-NAME          PIC X(50).

       PROCEDURE DIVISION.
       0000-MAIN.
           DISPLAY "================================================"
           DISPLAY " PORTVAL ENTITY CONTEXT QA SANITY CHECK        "
           DISPLAY "================================================"

      *> TC-01 SG baseline: currency, locale, regulator, centre, fee
           MOVE "TC-01 SG baseline core fields" TO WS-TEST-NAME
           MOVE "SG" TO WS-ENTITY-ID
           PERFORM 9000-LOAD-ENTITY
           IF WS-CURRENCY          = "SGD"
           AND WS-LOCALE           = "en_SG"
           AND WS-REGULATOR(1:3)   = "MAS"
           AND WS-BOOKING-CENTRE(1:9) = "Singapore"
           AND WS-MGMT-FEE-BPS     = 050
               PERFORM 9900-PASS
           ELSE
               DISPLAY "  GOT CURR=" WS-CURRENCY " LOC=" WS-LOCALE
                   " REG=" WS-REGULATOR " FEE=" WS-MGMT-FEE-BPS
               PERFORM 9910-FAIL
           END-IF

      *> TC-02 SG threshold=250000 + suitability=MAS_FAA_2002
           MOVE "TC-02 SG thresh=250000 suit=MAS_FAA_2002"
               TO WS-TEST-NAME
           MOVE "SG" TO WS-ENTITY-ID
           PERFORM 9000-LOAD-ENTITY
           IF WS-LARGE-POS-THRESH   = 000250000.00
           AND WS-SUITABILITY(1:12) = "MAS_FAA_2002"
               PERFORM 9900-PASS
           ELSE
               DISPLAY "  GOT THRESH=" WS-LARGE-POS-THRESH
                   " SUIT=" WS-SUITABILITY
               PERFORM 9910-FAIL
           END-IF

      *> TC-03 HK switch: HKD, SFC, fee=60, thresh=1000000
           MOVE "TC-03 HK currency=HKD reg=SFC fee=60 thresh=1000000"
               TO WS-TEST-NAME
           MOVE "HK" TO WS-ENTITY-ID
           PERFORM 9000-LOAD-ENTITY
           IF WS-CURRENCY          = "HKD"
           AND WS-LOCALE           = "en_HK"
           AND WS-REGULATOR(1:3)   = "SFC"
           AND WS-MGMT-FEE-BPS     = 060
           AND WS-LARGE-POS-THRESH = 001000000.00
           AND WS-SUITABILITY(1:12)= "SFC_COP_2019"
               PERFORM 9900-PASS
           ELSE
               DISPLAY "  GOT CURR=" WS-CURRENCY " FEE="
                   WS-MGMT-FEE-BPS " THRESH=" WS-LARGE-POS-THRESH
               PERFORM 9910-FAIL
           END-IF

      *> TC-04 CH switch: CHF, FINMA, fee=80, thresh=5000000
           MOVE "TC-04 CH currency=CHF reg=FINMA fee=80 thresh=5M"
               TO WS-TEST-NAME
           MOVE "CH" TO WS-ENTITY-ID
           PERFORM 9000-LOAD-ENTITY
           IF WS-CURRENCY          = "CHF"
           AND WS-LOCALE           = "de_CH"
           AND WS-REGULATOR(1:5)   = "FINMA"
           AND WS-MGMT-FEE-BPS     = 080
           AND WS-LARGE-POS-THRESH = 005000000.00
           AND WS-SUITABILITY(1:16)= "FINMA_LSFin_2020"
               PERFORM 9900-PASS
           ELSE
               DISPLAY "  GOT CURR=" WS-CURRENCY " FEE="
                   WS-MGMT-FEE-BPS " THRESH=" WS-LARGE-POS-THRESH
               PERFORM 9910-FAIL
           END-IF

      *> TC-05 Cross-entity: SG fee(50) NE HK fee(60) NE CH fee(80)
           MOVE "TC-05 Cross-entity fee-bps all distinct"
               TO WS-TEST-NAME
           IF 050 NOT = 060
           AND 060 NOT = 080
           AND 050 NOT = 080
               PERFORM 9900-PASS
           ELSE
               PERFORM 9910-FAIL
           END-IF

      *> TC-06 Vault refs use KEYVAULT pattern, no plaintext
      *>  (Static: verified by test_portval_static.py; noted here)
           MOVE "TC-06 Vault refs use KEYVAULT pattern (static ok)"
               TO WS-TEST-NAME
           IF WS-VAULT-REF(1:12) = "${KEYVAULT:w"
               PERFORM 9900-PASS
           ELSE
               DISPLAY "  GOT VAULT=" WS-VAULT-REF
               PERFORM 9910-FAIL
           END-IF

           DISPLAY "------------------------------------------------"
           DISPLAY " PASS: " WS-PASS-COUNT "  FAIL: " WS-FAIL-COUNT
           DISPLAY "------------------------------------------------"
           IF WS-FAIL-COUNT > ZERO
               STOP RUN RETURNING 1
           END-IF
           STOP RUN.

      *> -----------------------------------------------------------
      *> Dispatch mirroring PORTVAL 0100-LOAD-ENTITY-CONTEXT logic
      *> -----------------------------------------------------------
       9000-LOAD-ENTITY.
           EVALUATE WS-ENTITY-ID
               WHEN "SG"
                   MOVE "SGD"                   TO WS-CURRENCY
                   MOVE "en_SG"                 TO WS-LOCALE
                   MOVE "MAS     "              TO WS-REGULATOR
                   MOVE "Singapore       "      TO WS-BOOKING-CENTRE
                   MOVE 050                     TO WS-MGMT-FEE-BPS
                   MOVE 000250000.00            TO WS-LARGE-POS-THRESH
                   MOVE "MAS_FAA_2002        "  TO WS-SUITABILITY
                   MOVE "${KEYVAULT:wealth-sg-db-password}"
                                                TO WS-VAULT-REF
               WHEN "HK"
                   MOVE "HKD"                   TO WS-CURRENCY
                   MOVE "en_HK"                 TO WS-LOCALE
                   MOVE "SFC     "              TO WS-REGULATOR
                   MOVE "Hong Kong       "      TO WS-BOOKING-CENTRE
                   MOVE 060                     TO WS-MGMT-FEE-BPS
                   MOVE 001000000.00            TO WS-LARGE-POS-THRESH
                   MOVE "SFC_COP_2019        "  TO WS-SUITABILITY
                   MOVE "${KEYVAULT:wealth-hk-db-password}"
                                                TO WS-VAULT-REF
               WHEN "CH"
                   MOVE "CHF"                   TO WS-CURRENCY
                   MOVE "de_CH"                 TO WS-LOCALE
                   MOVE "FINMA   "              TO WS-REGULATOR
                   MOVE "Zurich          "      TO WS-BOOKING-CENTRE
                   MOVE 080                     TO WS-MGMT-FEE-BPS
                   MOVE 005000000.00            TO WS-LARGE-POS-THRESH
                   MOVE "FINMA_LSFin_2020    "  TO WS-SUITABILITY
                   MOVE "${KEYVAULT:wealth-ch-db-password}"
                                                TO WS-VAULT-REF
               WHEN OTHER
                   DISPLAY "FATAL: Unknown entity [" WS-ENTITY-ID "]"
                   STOP RUN RETURNING 99
           END-EVALUATE.

       9900-PASS.
           ADD 1 TO WS-PASS-COUNT
           DISPLAY "  PASS: " WS-TEST-NAME.

       9910-FAIL.
           ADD 1 TO WS-FAIL-COUNT
           DISPLAY "  FAIL: " WS-TEST-NAME.
